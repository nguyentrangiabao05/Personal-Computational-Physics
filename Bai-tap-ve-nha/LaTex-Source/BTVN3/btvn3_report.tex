\section{Nội dung bài tập}

Bài toán yêu cầu khảo sát dao động của con lắc đơn chiều dài $l$ với góc lệch $\theta$ so với phương thẳng đứng. Hai phương trình cần xét là
\begin{equation}
\theta'' + \frac{g}{l}\sin\theta = 0,
\label{eq:nonlinear}
\end{equation}
ứng với mô hình con lắc đầy đủ, và gần đúng góc nhỏ
\begin{equation}
\theta'' + \frac{g}{l}\theta = 0.
\label{eq:linear}
\end{equation}

Yêu cầu chính của bài tập:
\begin{enumerate}
    \item Hoàn thiện chương trình giải bài toán bằng ba phương pháp số: Euler thường, Euler cải tiến (MEM) và Runge--Kutta bậc 4 (RK4).
    \item Với $\theta_0 = \pi/6$, $\dot\theta_0 = 0$, $l = 60$ cm, vẽ và so sánh đồ thị $\theta(t)$ của các phương pháp số cho phương trình~\eqref{eq:nonlinear} và~\eqref{eq:linear}.
    \item So sánh nghiệm của phương trình~\eqref{eq:nonlinear} và~\eqref{eq:linear} cho hai điều kiện đầu $\theta_0 = \pi/12$ và $\theta_0 = \pi/4$; phần đồ thị này được vẽ bằng Gnuplot.
\end{enumerate}

\section{Cơ sở lý thuyết}

\subsection{Đưa phương trình bậc hai về hệ bậc một}

Đặt
\[
Y(t) = 
\begin{bmatrix}
\theta(t)\\
\omega(t)
\end{bmatrix},
\qquad \omega(t)=\dot\theta(t).
\]
Khi đó hai phương trình vi phân trở thành các hệ bậc một:
\begin{align}
\frac{d\theta}{dt} &= \omega,
&
\frac{d\omega}{dt} &= -\frac{g}{l}\sin\theta
\qquad \text{(PT [1])},
\label{eq:sys1}
\\
\frac{d\theta}{dt} &= \omega,
&
\frac{d\omega}{dt} &= -\frac{g}{l}\theta
\qquad \text{(PT [2])}.
\label{eq:sys2}
\end{align}

Trong bài này dùng $g = 9.81\ \mathrm{m/s^2}$ và $l = 0.6\ \mathrm{m}$, do đó tần số góc của mô hình góc nhỏ là
\[
\Omega = \sqrt{\frac{g}{l}} \approx 4.0435\ \mathrm{s^{-1}},
\]
và chu kỳ tương ứng là
\[
T_0 = 2\pi\sqrt{\frac{l}{g}} \approx 1.5539\ \mathrm{s}.
\]

Đối với phương trình tuyến tính~\eqref{eq:linear}, nghiệm giải tích với điều kiện đầu $\theta(0)=\theta_0$, $\omega(0)=\omega_0$ là
\begin{equation}
\theta(t)=\theta_0\cos(\Omega t)+\frac{\omega_0}{\Omega}\sin(\Omega t).
\label{eq:exact_linear}
\end{equation}
Trong trường hợp đang xét, $\omega_0=0$ nên
\[
\theta(t)=\theta_0\cos(\Omega t).
\]

\subsection{Các phương pháp số}

Giả sử $t_n=a+nh$, với bước $h=(b-a)/N$.

\paragraph{Phương pháp Euler thường.}
Với hệ $Y'=F(t,Y)$, công thức cập nhật là
\begin{equation}
Y_{n+1}=Y_n+hF(t_n,Y_n).
\label{eq:euler}
\end{equation}
Đây là phương pháp bậc một nên sai số tích lũy tương đối lớn, đặc biệt với bài toán dao động.

\paragraph{Phương pháp Euler cải tiến (Modified Euler Method -- MEM).}
Phương pháp này dùng trung bình giữa độ dốc ở đầu bước và độ dốc dự đoán ở cuối bước:
\begin{equation}
Y_{n+1}=Y_n+\frac{h}{2}\left[F(t_n,Y_n)+F\bigl(t_{n+1},Y_n+hF(t_n,Y_n)\bigr)\right].
\label{eq:mem}
\end{equation}
Phương pháp có bậc chính xác cao hơn Euler thường và ổn định hơn với dao động điều hòa.

\paragraph{Phương pháp Runge--Kutta bậc 4.}
Công thức cập nhật là
\begin{align}
K_1 &= hF(t_n,Y_n),\nonumber\\
K_2 &= hF\left(t_n+\frac{h}{2},Y_n+\frac{K_1}{2}\right),\nonumber\\
K_3 &= hF\left(t_n+\frac{h}{2},Y_n+\frac{K_2}{2}\right),\nonumber\\
K_4 &= hF(t_n+h,Y_n+K_3),\nonumber\\
Y_{n+1} &= Y_n+\frac{K_1+2K_2+2K_3+K_4}{6}.
\label{eq:rk4}
\end{align}
Đây là phương pháp bậc bốn, cho độ chính xác rất cao với cùng bước thời gian.

\section{Phân tích chương trình}

Chương trình được viết bằng Python trong notebook \texttt{daodongdieuhoa.ipynb}. Các hàm chính gồm:

\subsection{Khai báo hệ phương trình}
\begin{lstlisting}[style=mypython,caption={Khai báo hai hệ phương trình của bài toán con lắc}]
def phuongtrinh_1(t, Y):
    theta = Y[0]
    omega = Y[1]
    return np.array([omega, -(g/l)*np.sin(theta)])

def phuongtrinh_2(t, Y):
    theta = Y[0]
    omega = Y[1]
    return np.array([omega, -(g/l)*theta])
\end{lstlisting}

\textbf{Ý nghĩa.} 
\texttt{phuongtrinh\_1} biểu diễn mô hình con lắc phi tuyến đầy đủ, còn \texttt{phuongtrinh\_2} là gần đúng góc nhỏ. Cả hai đều được đưa về dạng vector $Y' = F(t,Y)$ để có thể dùng chung các bộ giải số.

\subsection{Hàm Euler thường}
\begin{lstlisting}[style=mypython,caption={Thuật toán Euler thường}]
def euler(a,b,N,alpha,phuongtrinh,filename):
    h = (b-a)/N
    x = [a]
    y = [alpha]

    for i in range(1, N+1):
        y.append(y[i-1] + h * phuongtrinh(x[i-1], y[i-1]))
        x.append(a + i*h)

    return x, y
\end{lstlisting}

\subsection{Hàm Euler cải tiến (MEM)}
\begin{lstlisting}[style=mypython,caption={Thuật toán Euler cải tiến}]
def Euler_modified(a,b,N,alpha,phuongtrinh,filename):
    h = (b-a)/N
    x = [a]
    y = [alpha]

    for i in range(1, N+1):
        x.append(a + i*h)
        y.append(y[i-1] + h/2 * (
            phuongtrinh(x[i-1], y[i-1])
            + phuongtrinh(x[i], y[i-1] + h*phuongtrinh(x[i-1], y[i-1]))
        ))

    return x, y
\end{lstlisting}

\subsection{Hàm RK4}
\begin{lstlisting}[style=mypython,caption={Thuật toán Runge--Kutta bậc 4}]
def RK4(a, b, N, alpha, phuongtrinh, filename):
    h = (b - a) / N
    x = [a]
    y = [alpha]

    for i in range(1, N + 1):
        x.append(a + i*h)

        k1 = h * phuongtrinh(x[i-1], y[i-1])
        k2 = h * phuongtrinh(x[i-1] + h/2, y[i-1] + k1/2)
        k3 = h * phuongtrinh(x[i-1] + h/2, y[i-1] + k2/2)
        k4 = h * phuongtrinh(x[i-1] + h,   y[i-1] + k3)

        y.append(y[i-1] + (k1 + 2*k2 + 2*k3 + k4)/6)

    return x, y
\end{lstlisting}

Ngoài việc tính nghiệm số, chương trình còn ghi dữ liệu $t$, $\theta$, $\omega$ ra các file \texttt{.txt}; những file này được dùng để vẽ đồ thị bằng Python và Gnuplot.

\section{Kết quả và thảo luận}

\end{itemize}

\subsection{So sánh PT [1] và PT [2] cho hai góc ban đầu khác nhau}

Ở phần này chỉ dùng RK4 để tránh ảnh hưởng của sai số phương pháp số, sau đó so sánh trực tiếp nghiệm của PT [1] và PT [2]. Hai trường hợp được xét là:
\[
\theta_0 = \frac{\pi}{12} \quad \text{và} \quad \theta_0 = \frac{\pi}{4},
\qquad \omega_0=0.

    \item Khi $\theta_0=\pi/4$, sự sai khác giữa hai mô hình tăng lên rõ rệt. Biên độ ban đầu vẫn giống nhau, nhưng PT [1] dao động chậm hơn PT [2], dẫn đến lệch pha tích lũy ngày càng lớn theo thời gian.
    \item Nói cách khác, với góc ban đầu đủ lớn, bài toán không còn có thể mô tả chính xác chỉ bằng mô hình góc nhỏ.
\end{itemize}

Để định lượng hơn, chu kỳ của PT [2] luôn bằng
\[
T_0 = 2\pi\sqrt{\frac{l}{g}} \approx 1.553893\ \mathrm{s}.
\]
Trong khi đó, chu kỳ của con lắc phi tuyến phụ thuộc vào biên độ và có thể viết dưới dạng tích phân elliptic hoàn chỉnh bậc nhất:
\[
T(\theta_0)=4\sqrt{\frac{l}{g}}\,K\!\left(\sin^2\frac{\theta_0}{2}\right).
\]

    \item Với $\theta_0=\pi/4$, chu kỳ phi tuyến lớn hơn khoảng $4.0\%$. Sai lệch này đủ lớn để tạo ra lệch pha đáng kể sau nhiều chu kỳ, đúng như quan sát trên đồ thị.

Về mặt vật lý, kết quả này xuất phát từ việc gần đúng $\sin\theta \approx \theta$ chỉ đúng tốt khi $|\theta| \ll 1$. Khi biên độ tăng, thành phần phi tuyến làm lực kéo về không còn tỉ lệ tuyến tính với $\theta$, do đó chu kỳ phụ thuộc vào biên độ và trở nên lớn hơn giá trị góc nhỏ.

\section{Kết luận}

Từ các kết quả tính toán và so sánh ở trên, có thể rút ra các kết luận sau:
\begin{enumerate}
    \item Việc chuyển phương trình con lắc về hệ bậc một cho phép áp dụng trực tiếp các phương pháp số chuẩn như Euler, MEM và RK4.
    \item Với cùng bước thời gian $h=0.01$ s, Euler thường cho sai số lớn và làm biên độ tăng dần theo thời gian, nên không phù hợp để mô phỏng dao động trong thời gian dài.
    \item MEM cải thiện đáng kể so với Euler thường; nghiệm thu được bám khá tốt lời giải đúng của mô hình góc nhỏ.
    \item RK4 cho kết quả tốt nhất trong ba phương pháp, với sai số gần như không đáng kể trong bài toán này.
    \item Gần đúng góc nhỏ PT [2] mô tả rất tốt con lắc khi $\theta_0=\pi/12$, nhưng bắt đầu sai khác rõ rệt khi $\theta_0=\pi/4$. Sai khác chủ yếu thể hiện ở sự tăng chu kỳ và lệch pha của mô hình phi tuyến PT [1].
\end{enumerate}

Như vậy, nếu chỉ cần khảo sát dao động biên độ nhỏ thì phương trình tuyến tính là đủ. Tuy nhiên, khi góc lệch ban đầu không còn nhỏ, cần dùng phương trình phi tuyến đầy đủ và một phương pháp số có độ chính xác cao, trong đó RK4 là lựa chọn phù hợp nhất trong phạm vi bài tập này.
