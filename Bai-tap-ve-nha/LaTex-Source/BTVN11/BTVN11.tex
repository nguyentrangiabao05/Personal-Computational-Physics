\section{Cơ sở lý thuyết}

\subsection{Phương trình Schrödinger phụ thuộc thời gian}

Phương trình Schrödinger phụ thuộc thời gian (TDSE) một chiều cho hàm sóng $\psi(x,t)$ trong thế năng $V(x)$:
\begin{equation}
i\hbar\,\frac{\partial \psi(x,t)}{\partial t} = -\frac{\hbar^2}{2m}\,\frac{\partial^2 \psi(x,t)}{\partial x^2} + V(x)\,\psi(x,t).
\label{eq:tdse}
\end{equation}
Trong đơn vị tự nhiên $\hbar=m=1$ dùng trong bài, đại lượng được bảo toàn theo thời gian là điều kiện chuẩn hóa
\begin{equation}
N(t) = \int_{-\infty}^{\infty} |\psi(x,t)|^2\,dx = \text{const}.
\label{eq:norm}
\end{equation}

\subsection{Sai phân hữu hạn hiện (Euler tiến / FTCS)}

Rời rạc hóa miền không gian--thời gian với bước lưới $h=\Delta x$, $k=\Delta t$, $\psi^n_j \equiv \psi(x_j,t_n)$. Đạo hàm thời gian được xấp xỉ bằng sai phân tiến (Euler tiến), đạo hàm không gian bằng sai phân trung tâm bậc hai:
\begin{equation}
\frac{\partial \psi}{\partial t}\bigg|_{j}^{n} \approx \frac{\psi^{n+1}_j-\psi^{n}_j}{k}, \qquad
\frac{\partial^2 \psi}{\partial x^2}\bigg|_{j}^{n} \approx \frac{\psi^{n}_{j+1}-2\psi^{n}_j+\psi^{n}_{j-1}}{h^2}.
\end{equation}
Thay vào \eqref{eq:tdse} và giải $\psi^{n+1}_j$ ta được công thức cập nhật \emph{hiện} (explicit, FTCS):
\begin{equation}
\psi^{n+1}_j = \underbrace{\left[1 - i\frac{k}{\hbar}V_j - i\frac{\hbar k}{m h^2}\right]}_{g_j}\psi^{n}_j
+ \underbrace{i\frac{\hbar k}{2mh^2}}_{a}\left(\psi^{n}_{j+1}+\psi^{n}_{j-1}\right).
\label{eq:euler-update}
\end{equation}
Công thức này được đặt trong hàm \texttt{euler\_giai\_psi}, với $a$ và $g_j$ (biến \texttt{g\_x}) như trong \eqref{eq:euler-update}.

\section{Thực hành 1 -- Mô phỏng bó sóng Gauss bằng Euler tiến}

\subsection{Thiết lập bài toán}

Điều kiện đầu là một gói sóng Gauss có động lượng trung bình $k_0$:
\begin{equation}
\psi(x,0) = \exp\!\left[-\frac{1}{2}\left(\frac{x-5}{\sigma_0}\right)^2\right] e^{ik_0 x}, \qquad \sigma_0=0.5,\quad k_0 = 17\pi.
\end{equation}
Chuẩn hóa ban đầu có thể tính giải tích: $N(0)=\int |\psi(x,0)|^2\,dx = \sigma_0\sqrt{\pi} \approx 0.8862$, đúng với giá trị đầu tiên trong bảng kết quả (Bảng~\ref{tab:norm}).

Thế năng $V(x)$ được khai báo bằng giá trị $V_0=0$ trên toàn bộ khoảng $[0,15]$ và $V=100$ chỉ tại đúng hai điểm biên $x=0,\,15$:
\begin{equation}
V(x) =
\begin{cases}
0, & 0 \le x \le 15,\\
100, & x < 0 \text{ hoặc } x > 15.
\end{cases}
\end{equation}

Tham số lưới: $\Delta x = 0.02$, $\Delta t = \tfrac{1}{2}\Delta x^2 = 2\times10^{-4}$ (giống điều kiện Courant của bài toán truyền nhiệt), $t\in[0,\,0.5]$.

\subsection{Phương pháp Euler tiến}

\begin{mintedbox}{python}
def khoitao_psi0(x):
    global sigma_0, k_0
    psi_0 = np.exp(-1/2 * ((x-5)/sigma_0)**2 )*np.exp(1j*k_0*x)
    return psi_0

def thenang(x):
    N_x = len(x)
    V = np.zeros(N_x)
    for i in range(N_x):
        if x[i] <= 0 or x[i] >= 15:
            V[i] = 0
        else:
            V[i] = 100
    return V

def euler_giai_psi(x, t):
    global hbar, m, dx, dt, N_x, N_t
    psi_t_x = np.zeros([N_t,N_x], dtype=np.complex128)
    psi_0 = khoitao_psi0(x)
    V = thenang(x)

    a = 1j * hbar * dt / (2*m*dx**2)
    g_x = 1 - 1j*dt/hbar * V - 1j*hbar*dt/(m*dx**2)

    psi_t_x[0] = psi_0
    for n in range(0, N_t-1):
        for j in range(1, N_x-1):
            psi_t_x[n+1,j] = g_x[j] * psi_t_x[n,j] + a*(psi_t_x[n,j+1]+psi_t_x[n,j-1])

    return psi_t_x
\end{mintedbox}

Hàm \texttt{luu\_file\_xac\_xuat} tính chuẩn hóa $N(t)=\int |\psi(x,t)|^2\,dx$ bằng quy tắc hình thang (\texttt{np.trapz}) tại mỗi mốc thời gian được lưu (\texttt{save\_every=50}, tức mỗi $\Delta t \times 50 = 0.01$).

\section{Kết luận}

\begin{itemize}[leftmargin=*]
\item Phương pháp Euler tiến cho phương trình Schrödinger phụ thuộc thời gian là \textbf{bất ổn định}.
\item Kết quả số cho thấy chuẩn hóa $N(t)$ tăng rất nhanh, từ khoảng $0.886$ lên hơn $10^{295}$ và xảy ra tràn số tại $t\approx 0.22$.
\item Vì vậy, Euler tiến \textbf{không phù hợp} để giải TDSE; nên sử dụng Crank--Nicolson hoặc phương pháp split-step Fourier.

\end{itemize}
